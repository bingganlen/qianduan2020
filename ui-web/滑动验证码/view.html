;(function (w) {
var timingLog = {};
var prefix = '^';

var timing = {
log: function (type) {
var cur = +new Date();

if (typeof type === 'string') {
timingLog[prefix + type] = cur;
timingLog[type] = null;
} else {
for (var i = 0; i < type.length; i++) {
timingLog[prefix + type[i]] = cur;
timingLog[type[i]] = null;
}
}
},
logEnd: function (type) {
// 不覆盖
if (timingLog[type]) return timingLog[type];

var cur = +new Date();
timingLog[type] = cur - (timingLog[prefix + type] || cur);

return timingLog[type];
},
point: function (type) {
return timingLog[prefix + type] || +new Date();
}
};

w.timing = timing;
})(window);
